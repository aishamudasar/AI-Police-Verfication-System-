import cv2
import mediapipe as mp
import numpy as np
import time

# Initialize MediaPipe
mp_face_mesh = mp.solutions.face_mesh
face_mesh = mp_face_mesh.FaceMesh()

# Eye landmarks for blink detection
LEFT_EYE = [33, 160, 158, 133, 153, 144]
RIGHT_EYE = [362, 385, 387, 263, 373, 380]

# Store registered face embedding (simulated database)
registered_embedding = None

def eye_aspect_ratio(landmarks, eye_indices, w, h):
    points = []
    for idx in eye_indices:
        x = int(landmarks[idx].x * w)
        y = int(landmarks[idx].y * h)
        points.append((x, y))

    vertical = abs(points[1][1] - points[5][1])
    horizontal = abs(points[0][0] - points[3][0])

    return vertical / horizontal if horizontal != 0 else 0

# Create embedding (SIMULATION of FaceNet)
def get_embedding(landmarks):
    embedding = []
    for lm in landmarks:
        embedding.append(lm.x)
        embedding.append(lm.y)
    return np.array(embedding)

# Euclidean distance matching
def match_faces(emb1, emb2):
    dist = np.linalg.norm(emb1 - emb2)
    return dist

cap = cv2.VideoCapture(0)

blink_counter = 0
blink_detected = False
registered = False

print("Press 'r' to REGISTER face")
print("Press 'v' to VERIFY face")

while True:
    ret, frame = cap.read()
    if not ret:
        break

    h, w, _ = frame.shape
    rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    results = face_mesh.process(rgb)

    if results.multi_face_landmarks:
        for face_landmarks in results.multi_face_landmarks:
            landmarks = face_landmarks.landmark

            # --------- LIVENESS (Blink Detection) ----------
            left_ear = eye_aspect_ratio(landmarks, LEFT_EYE, w, h)
            right_ear = eye_aspect_ratio(landmarks, RIGHT_EYE, w, h)
            avg_ear = (left_ear + right_ear) / 2

            if avg_ear < 0.2:
                blink_counter += 1
            else:
                if blink_counter > 2:
                    blink_detected = True
                blink_counter = 0

            # --------- FACE EMBEDDING ----------
            embedding = get_embedding(landmarks)

            # --------- DISPLAY ----------
            if blink_detected:
                cv2.putText(frame, "Liveness: VERIFIED", (30, 50),
                            cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
            else:
                cv2.putText(frame, "Blink to verify...", (30, 50),
                            cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 255), 2)

    cv2.imshow("AI Verification System", frame)

    key = cv2.waitKey(1)

    # --------- REGISTER FACE ----------
    if key == ord('r') and results.multi_face_landmarks:
        registered_embedding = embedding
        registered = True
        print("Face Registered Successfully")

    # --------- VERIFY FACE ----------
    if key == ord('v') and registered and blink_detected:
        distance = match_faces(registered_embedding, embedding)

        print("Matching Distance:", distance)

        if distance < 5:  # threshold (adjustable)
            print("IDENTITY VERIFIED ✅")
        else:
            print("IDENTITY NOT MATCHED ❌")

    # Exit
    if key == 27:
        break

cap.release()
cv2.destroyAllWindows()
